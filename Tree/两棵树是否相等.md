### 两棵树是否相等

```javaScript
let obj = {
  val: 1,
  left: {
    val: 2,
    left: {
      val: 4,
      left: null,
      right: null
    },
    right: {
      val: 5,
      left: null,
      right: null
    },
  },
  right: {
    val: 2,
    left: {
      val: 5,
      left: null,
      right: null
    },
    right: {
      val: 4,
      left: null,
      right: null
    }
  }
}
```
1、递归 

```javaScript
function sameTree(p, q) {
  if (!p && !q) { return true }
  if ((p && !q) || (!p && q) || (p.val !== q.val)) { return false }
  return sameTree(p.left, q.left) && sameTree(p.right, q.right)
}
```

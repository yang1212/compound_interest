1、递归

```javaScript
function revertTree(root) {
  if (root.length === 0) { return '' }
  const travse = function(root) {
    let temp = root.left
    root.left = root.right
    root.right = temp
    root.left && travse(root.left)
    root.right && travse(root.right)
  }
  travse(root)
  return root
}
```

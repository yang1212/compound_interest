### 先序(中左右)

````javaScript
let obj = {
  val: 1,
  left: {
    val: 2,
    left: null,
    right: null
  },
  right: {
    val: 3,
    left: null,
    right: null
  }
}
````


1、 递归

````javaScript
function frontTravseDG(root) {
  const result = []
  const frontTravseNode = function(root) {
    if (root !== null) {
      result.push(root.val)
      frontTravseNode(root.left)
      frontTravseNode(root.right)
    }
  }
  frontTravseNode(root)
  return result
}
````


2、迭代
* Attention: 栈的特性：后入先出

````javaScript
function frontTravseDD(root) {
  let result = []
  let stack = []
  const travse = function(root) {
    if (root !== null) { stack.push(root)}
    while(stack.length) {
      let temp = stack.pop()
      result.push(temp.val)
      if (temp.right) { stack.push(temp.right) }
      if (temp.left) { stack.push(temp.left) }
    }
  }
  travse(root)
  return result
}
````
